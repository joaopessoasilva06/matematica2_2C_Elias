//Faça um codigo que leia uma cor. Se ela for azul, diga: cor fria. se for vermelha, diga: cor quente. se for verde, diga: cor natural. Se for outra cor, diga: cor invalida.

    console.log("cor fria");
}else if(cor === "vermelha"){
    console.log("cor quente")
}else if(cor === "verde"){
    console.log("cor natural");
}else{
